function graphQLFetcher(graphQLParams) { // This example expects a GraphQL server at the path /graphql. // Change this to point wherever you host your GraphQL server. return fetch('/library', { method: 'post', headers: { 'Accept': 'application/json', 'Content-Type': 'application/json', }, body: JSON.stringify(graphQLParams), credentials: 'include', }).then(function (response) { return response.text(); }).then(function (responseBody) { try { return JSON.parse(responseBody); } catch (error) { return responseBody; } }); }
